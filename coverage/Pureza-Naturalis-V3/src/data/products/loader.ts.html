/**
 * Normaliza textos con caracteres mal codificados (mojibake) comunes en espaol.
 * Se aplica a propiedades de texto conocidas del producto.
 */
export function normalizeProductText(p: Product): Product {
  const decodeLatin1Utf8 = (s?: string): string => {
    const str = s ?? '';
    if (!str) return '';
    try {
      if (/[']||||/.test(str)) {
        // Heurstica: texto con mojibake tpico ISO-8859-1 interpretado como UTF8
        // escape -> convierte a %u00XX (latin1); decodeURIComponent interpreta como UTF8
        // Deprecated pero disponible en navegadores; fallback seguro si falla
        return decodeURIComponent(escape(str));
      }
      return str;
    } catch {
      return str;
    }
  };
 
  // Reparacin adicional de mojibake comn (, , , ), con hasta 2 pasadas
  const decodeUtf8Mojibake = (s?: string): string => {
    let current = s ?? '';
    if (!current) return '';
    for (let i = 0; i < 2; i++) {
      try {
        if (!/[\uFFFD]/.test(current)) break;
        const decoded = decodeURIComponent(escape(current));
        if (decoded === current) break;
        current = decoded;
      } catch {
        break;
      }
    }
    return current;
  };
 
  const fix = (s?: string): string => {
    let t = s ?? '';
    // Intenta reparar varias capas de mojibake
    t = decodeUtf8Mojibake(decodeLatin1Utf8(t));
    // Espacios repetidos y limpieza ligera
    t = t
      .replace(/[\u0300-\u036f]/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();
    return t;
  };
 
  const clone = { ...p } as Product & Record<string, unknown>;
  const fields = [
    'name',
    'description',
    'detailedDescription',
    'mechanismOfAction',
    'dosage',
    'administrationMethod',
  ];
  fields.forEach((k) => (clone[k] = fix(clone[k] as string | undefined)));
 
  if (Array.isArray(clone.benefitsDescription)) {
    clone.benefitsDescription = clone.benefitsDescription.map((s: string) =>
      fix(s)
    );
  }
  if (Array.isArray(clone.healthIssues)) {
    clone.healthIssues = clone.healthIssues.map((s: string) => fix(s));
  }
  if (Array.isArray(clone.tags)) {
    clone.tags = clone.tags.map((s: string) => fix(String(s)));
  }
  if (Array.isArray(clone.categories)) {
    clone.categories = clone.categories.map((s: string) => fix(String(s)));
  }
  if (Array.isArray((clone as Product).faqs)) {
    (clone as Product).faqs = ((clone as Product).faqs as FaqItem[]).map((f) => ({
      ...f,
      question: fix(f.question ?? f.title),
      answer: fix(f.answer),
    }));
  }
  if (clone.scientificReferences && Array.isArray(clone.scientificReferences)) {
    const normalizeStudyType = (s?: string) => {
      const base = (s || '').toString().toLowerCase().trim();
      const ascii = base.normalize('NFD').replace(/[\u0300-\u036f]/g, '');
      const hasAll = (arr: string[]) => arr.every((k) => ascii.includes(k));
      if (hasAll(['ensayo', 'clinic'])) return 'ensayo-clinico';
      if (hasAll(['meta', 'anal'])) return 'meta-analisis';
      if (hasAll(['revision', 'sist'])) return 'revision-sistematica';
      if (ascii.includes('revision')) return 'revision-narrativa';
      if (ascii.includes('guia') || ascii.includes('guideline'))
        return 'guia-clinica';
      if (ascii.includes('position') || ascii.includes('stand'))
        return 'position-stand';
      if (ascii.includes('animal')) return 'estudio-animal';
      if (ascii.includes('opinion') || ascii.includes('efsa'))
        return 'opinion-regulatoria';
      return base.replace(/\s+/g, '-');
    };
 
    type SciRef = NonNullable<Product['scientificReferences']>[number];
    clone.scientificReferences = (clone.scientificReferences as unknown[])
      .map((r: unknown) => {
        const ref = r as Record<string, unknown>;
        return {
          ...(ref as unknown as SciRef),
          title: fix(ref.title as string),
          journal: fix(ref.journal as string),
          summary: fix(ref.summary as string),
          studyType: normalizeStudyType(ref.studyType as string) as SciRef['studyType'],
          year: isNaN(parseInt(String(ref.year), 10))
            ? (ref.year as number)
            : parseInt(String(ref.year), 10),
        };
      })
      .sort((a: SciRef, b: SciRef) => {
        const ay =
          typeof a.year === 'number'
            ? a.year
            : parseInt(String(a.year), 10) || 0;
        const by =
          typeof b.year === 'number'
            ? b.year
            : parseInt(String(b.year), 10) || 0;
        return by - ay;
      });
  }
  return clone as Product;
}
